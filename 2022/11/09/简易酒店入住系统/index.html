    public static boolean rightPhoneNumber(String number) {
        for (int i = 0; i < number.length(); i++) {
            char c = number.charAt(i);
            if (c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z' || number.length() != 11 || number.charAt(0) != '1') {
                return false;
            }
        }
        return true;
    }
