一个猜数字的小游戏（雾

大概就是系统会生成随机的正确和错误的数字，你输入一个特定范围的数字，来判断你是否选中正确的数字

选中了正确的数字，到达一定次数会判断胜利

选了错误的数字，系统会从剩下的数字里随机挑选两个数字，一个正确的一个错误的，然后你继续判断输入

当错误次数到达一定次数，也会自动判断失败

如果当所有数字都输完了，结束的时候也有个判断，符合后也会判断失败

默认是猜9(index)位数字，其中正确数字(index+1)/2 (right)位，错误(index-1)/2 (error)位

当正确次数达到 (right)+1/2 或者错误次数 (error)+2/2 判断成功或失败

输入完所有数字如果正确次数小于等于 (right)-1/2 或者 错误次数等于0次 也会判定失败

    public static void start{
// 创建一个数组，用来存储表示正确与错误的位置，用0代表错误，1为正确
    int[] arr = new int[9];
        
// 先设置一共猜多少个数字，默认1-9，数组长度可以改成非偶数的长度
    int[] arr1 = new int[arr.length];
    indexArr(arr1);   

// 设置存储正确与错误数字的数组        
    int[] right = new int[(arr.length + 1) / 2];
    int[] error = new int[(arr.length - 1) / 2];
    
    // 定义判断变量
    int tcount = 0;
    int wcount = 0;
    // 将0或1按照随机索引存储到数组中，然后将其也存储到正确和错误的数组里
    for (int i = 0; i < arr.length; i++) {
            if (tcount != right.length) {
                arr[arr1[i] - 1] = 1;
                right[i] = arr1[i];
                tcount++;
            } else {
                arr[arr1[i] - 1] = 0;
                error[i - right.length] = arr1[i];
                wcount++;
            }
    }
   // 把存储到正确和错误数组里的数字进行冒牌排序
   newArr(error);
   newArr(right);
        
   //  定义记录成功和失败次数的变量    
   int wc = 0;
   int tc = 0;
   // 跳出循环条件
   boolean flag = true;
   // 定义两个列表，分别用来判断输入的数字是否重复和提示剩余数字
   ArrayList<Integer> list = new ArrayList<>();
   ArrayList<Integer> newList = new ArrayList<>();
   //  接下来每输入一位数字，都会将输入的数字和未输入的数字进行一次判断，然后将没有输入的数字进行添加到列表中用来提示信息
   for (int i = 0; i < arr1.length; i++) {
            newList.add(arr1[i]);
   }
        
    // 开始输入数字    
    for (int i = 0; i < arr.length; i++) {
            Scanner sc = new Scanner(System.in);
            while (true) {
                System.out.println("-----------------------------------------");
                System.out.println("请输入1-" + (arr.length) + "号位");
                System.out.println("当前是执行第" + (i + 1) + "次");
                System.out.println("-----------------------------------------");
                int inputNumber = sc.nextInt();
                // 判断输入数字是否重复
                boolean b = checkArr(list, inputNumber);
                if (b) {
                    // 将输入的数字添加到列表中，方便接下来的操作
                    list.add(inputNumber);
                    
                    // 输入了错误的数字，会调用一个notice方法
                    if (arr[inputNumber - 1] == 0) {
                        // 华丽的分隔符
                       System.out.println("///////////////////////////////////////////");
                        System.out.println("危险1次");
                        wc++;
                        tc--;
                        System.out.println("当前wc位" + wc);
                        System.out.println("当前tc位" + tc);
                        // 提示还可以输入的数字
                        checkList(list, arr1, newList);
                        if (wc != (error.length+2)/2) {
                        // 提示一个正确和错误的数字
                            notice(right, error, newList);
                        }
                       System.out.println("///////////////////////////////////////////");
                    } else if (arr[inputNumber - 1] == 1) {
                       System.out.println("///////////////////////////////////////////");
                        System.out.println("安全1次");
                        tc++;
                        wc--;
                        System.out.println("当前wc位" + wc);
                        System.out.println("当前tc位" + tc);
                        // 提示还可以输入的数字
                        checkList(list, arr1, newList);
                       System.out.println("///////////////////////////////////////////");
                    }

                    // 胜利或失败条件
                    if (tc == (right.length+1)/2) {
                        System.out.println("您成功了！");
                        flag = false;
                        break;
                    } else if (wc == (error.length+2)/2) {
                        System.out.println("您失败了");
                        flag = false;
                        break;
                    }
                    break;
                } else {
                    System.out.println("数字重复请重新输入");
                    // 同样提示剩余数字
                    checkList(list, arr1, newList);
                }
            }
          // 失败自动跳出循环
            if (!flag) {
                break;
            }
        }
      
        // 输入完所有的数字判断失败条件
        if (flag) {
            if (tc <= (right.length-1)/2 || wc == 0) {
                System.out.println("你失败了");
            }
        }
   
        
}

随机打乱索引值，方便判断也方便赋值

//  将数组索引打乱  
   public static void indexArr(int[] arr) {
        Random r = new Random();
        int key = r.nextInt(arr.length) + 1;
        arr[0] = key;
        for (int i = 0; i < arr.length; i++) {
            key = r.nextInt(arr.length) + 1;
            for (int j = 0; j < i; j++) {
                while (arr[j] == key) {
                    key = r.nextInt(arr.length) + 1;
                    j = 0;
                }
            }
            arr[i] = key;
        }
    }

主要是防止提示信息偶然的出现规律，其实也可以不加这条

//  将存储到正确和失败数组里的数字进行冒泡排序
    public static int[] newArr(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            for (int j = 0; j < arr.length - i - 1; j++) {
                if (arr[j + 1] > arr[j]) {
                    int temp = arr[j + 1];
                    arr[j + 1] = arr[j];
                    arr[j] = temp;
                }
            }
        }
        return arr;
    }

//  判断输入数字是否重复
    public static boolean checkArr(ArrayList<Integer> list, int number) {
        for (int i = 0; i < list.size(); i++) {
            Integer integer = list.get(i);
            if (integer == number) {
                return false;
            }
        }
        return true;
    }

 public static void checkList(ArrayList<Integer> list, int[] arr, ArrayList<Integer> 	  newList) {
         // 如果输入的数字存在，就把其去除
        Integer index = list.get(list.size() - 1);
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == index) {
                newList.remove(index);
            }
        }
        if (list.size() != arr.length) {
            System.out.print("当前还剩：");
            for (int i = 0; i < newList.size(); i++) {
                System.out.print(newList.get(i) + "\t");
            }
            System.out.println();
        } else {
            System.out.println("您已无机会");
        }
    }

 // 将未输入的数字筛选，然后提示一个正确和错误的数字
    public static void notice(int[] right, int[] error, ArrayList<Integer> newList) {
        int[] notice = new int[2];
        Random r = new Random();
        // 防止正确数字和错误数字出现的位置有规律
        int index = r.nextInt(2);
// 将剩余错误和正确的数字随机各挑选一个用作提示信息
        if (index == 1) {
            for (int i = 0; i < newList.size(); i++) {
                for (int j = 0; j < error.length; j++) {
                    if (newList.get(i) == error[j]) {
                        notice[0] = newList.get(i);
                    }
                }
            }

            for (int i = 0; i < newList.size(); i++) {
                for (int j = 0; j < right.length; j++) {
                    if (newList.get(i) == right[j]) {
                        notice[1] = newList.get(i);
                    }
                }
            }
        } else {
            for (int i = 0; i < newList.size(); i++) {
                for (int j = 0; j < right.length; j++) {
                    if (newList.get(i) == right[j]) {
                        notice[1] = newList.get(i);
                    }
                }
            }

            for (int i = 0; i < newList.size(); i++) {
                for (int j = 0; j < error.length; j++) {
                    if (newList.get(i) == error[j]) {
                        notice[0] = newList.get(i);
                    }
                }
            }
        }

        System.out.println("其中有一数字为正确，请选择,如果出现0，说明已无错误或正确数字");
        for (int i = 0; i < notice.length; i++) {
            System.out.print(notice[i] + "\t");
        }
        System.out.println();
    }
