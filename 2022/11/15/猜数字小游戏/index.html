   public static void indexArr(int[] arr) {
        Random r = new Random();
        int key = r.nextInt(arr.length) + 1;
        arr[0] = key;
        for (int i = 0; i < arr.length; i++) {
            key = r.nextInt(arr.length) + 1;
            for (int j = 0; j < i; j++) {
                while (arr[j] == key) {
                    key = r.nextInt(arr.length) + 1;
                    j = 0;
                }
            }
            arr[i] = key;
        }
    }
