获取一段字符串，给他转换成人看不懂的样子

没什么逻辑，完全瞎写

public class demo {
    public static void main(String[] args) {
        Random r = new Random();
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入原始密码");
        String def = sc.next();;

        // 第一步，转换成ascll
        ArrayList<Integer> OneStr = new ArrayList<>();
        for (int i = 0; i < def.length(); i++) {
            char c = def.charAt(i);
            OneStr.add((int) c);
        }


        // 第二部，随机增加数字
        ArrayList<Integer> once = once(OneStr);


        // 生成一些字符
        ArrayList<String> two = two(once);


        //转换成字符串
        StringBuilder sb = new StringBuilder();
        String str3 = "";
        for (int i = 0; i < once.size(); i++) {
            Integer temp = once.get(i);
            str3 = sb.append(temp).toString();
        }

        // 第三步，截取一部分乘以随机数然后重新设置
        for (int i = 0; i < OneStr.size(); i++) {
            int r1 = r.nextInt(5) + 1;
            char c = str3.charAt(i);
            int threeint = c * r1;
            OneStr.set(i,threeint);
        }


        // 将生成的所有字符拼接成字符串
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < two.size(); i++) {
            String s = two.get(i);
            result.append(s);
        }
        for (int i = 0; i < OneStr.size(); i++) {
            Integer in = OneStr.get(i);
            result.append(in);
        }
        char[] words = words();
        for (int i = 0; i < OneStr.size()+3; i++) {
            int index = r.nextInt(words.length);
            result.append(words[index]);
        }
        String fin = result.toString();


        // 随机打乱成最终效果
        String[] Str = new String[fin.length()];//最终字符存储的数组
        int[] rando = new int[fin.length()];//传递长度
        int[] newArr1 = isexist(rando);//索引值

        for (int i = 0; i < newArr1.length; i++) {
            char c = fin.charAt(i);
                Str[newArr1[i]] = String.valueOf(c);
        }
        for (int i = 0; i < Str.length; i++) {
            System.out.print(Str[i]);
        }

    }

    public static ArrayList<Integer> once(ArrayList<Integer> OneStr) {
        Random r = new Random();
        for (int i = 0; i < OneStr.size(); i++) {
            int rNumber = r.nextInt(100) + 1;
            OneStr.set(i,OneStr.get(i) + rNumber);
        }
        return OneStr;
    }

    public static ArrayList<String> two(ArrayList<Integer> once) {
        Random r = new Random();
        String[] ranStr = {"$", "*", "@", "!", "&", "%", "^", "]","{","?", "_","！", ";", "*", "……", "-", "@", "￥", "|", "#"};
        ArrayList<String> two = new ArrayList<>();
        for (int i = 0; i < once.size()*2; i++) {
            int index = r.nextInt(ranStr.length);
            two.add(ranStr[index]);
        }
        return two;
    }

    public static char[] words(){
        char[] arr = new char[52];
        for (int i = 0; i < arr.length; i++) {
            if(i>=0 && i<26){
                arr[i] = (char)(65+i);
            }else if(i>=26 && i<52){
                arr[i] = (char)(97+i-26);
            }
        }
        return arr;
    }

    public static int[] isexist(int[] ran) {
        Random r = new Random();
        ran[0] = r.nextInt(ran.length);
        for (int i = 1; i < ran.length; i++) {
            int index = r.nextInt(ran.length);
            for (int k = 0; k < i; k++) {
                while (ran[k] == index) {
                    index = r.nextInt(ran.length);
                    k = 0;
                }
            }
            ran[i] += index;
        }
        return ran;
    }
}
